```ruby
def hanoi(disk)
  $a= Array.new()
  $b= Array.new()
  $c= Array.new()
  j=disk
  for i in 0..disk-1
    $a[i]=j
    j-=1
  end
  sayac=0
  k = kontrol(disk)
  while k==false
    if $a.last==1
      $c << ($a.last)
      $a.pop
      puts "Move disk #{$c.last} from A to C"
      sayac +=1
    end
    k = kontrol(disk)
    break if k==true

    unless ($b.count>0 ? ($a.count > 0 ? ($b.last < $a.last) : true) :false)
      $b << ($a.last)
      $a.pop
      puts "Move disk #{$b.last} from A to B"
      sayac +=1
    else 
      $a << ($b.last)
      $b.pop
      puts "Move disk #{$b.last} from B to A"
      sayac +=1
    end

    $b << ($c.last)
    $c.pop
    puts "Move disk #{$b.last} from C to B"
    sayac +=1

    k = kontrol(disk)
    break if k==true
  
    if $a.empty? || ($c.count > 0 ? ($c.last < $a.last) : false)
        $a << ($c.last)
        $c.pop
        puts "Move disk #{$a.last} from C to A"
        sayac +=1
    else
      $c<<($a.last)
      $a.pop
      puts "Move disk #{$c.last} from A to C"
      sayac +=1
    end
  
    $a<<$b.last
    $b.pop
    puts "Move disk #{$a.last} from B to A"
    sayac +=1

    if ($b.last < $c.last) 
      $c << ($b.last)
      $b.pop
      puts "Move disk #{$c.last} from B to C"
      sayac +=1
    else
      $b<<($c.last)
      $c.pop
      puts "Move disk #{$b.last} from C to B"
      sayac +=1
    end
  end
puts sayac
puts "\nA#{$a}"
puts "B#{$b}"
puts "C#{$c}"
end

def kontrol(disk)
  for i in 1..disk-1
    unless $c.empty? || $c.length<disk
      if $c[i] < $c[i-1]
        status=true
      end
    else
      status=false
      break
    end    
  end
  if status==true then
    return status
  end
  for i in 1..disk-1
    unless $b.empty? || $b.length<disk
      if $b[i] < $b[i-1]
        status=true
      end
    else
      status=false
      break
    end    
  end
  return status
end
```

```ruby
def hanoi(n, source, auxiliary, target)
  if n > 0
    hanoi(n - 1, source, target, auxiliary)
    puts "Move disk #{n} from #{source} to #{target}"
    hanoi(n - 1, auxiliary, source, target)
  end
end
hanoi(10, 'A', 'B', 'C')
```
